\newcommand{\ficherosBasicosTeXiS}{%
TeXiS/TeXiS_pream,TeXiS/TeXiS_cab,TeXiS/TeXiS_bib,TeXiS/TeXiS_cover,%
TeXiS/TeXiS_part%
}

constantes,guionado,\proyecto/Cascaras/bibliografia,config%
}

}

\newcommand{\compilaApendice}[1]{%

}
